  <h1>üöñ Best Order Picker</h1>
  
  <label>Price (EGP): <input type="number" id="price" /></label>
  <label>Distance (km): <input type="number" id="distance" step="0.1" /></label>
  <label>Time (minutes): <input type="number" id="time" /></label>
  
  <button onclick="calculate()">Calculate</button>
  
  <div id="output" class="result"></div>
  
  <script>
    const minPerKm = 5;
    const minPerMin = 2;

    function calculate() {
      const price = parseFloat(document.getElementById('price').value);
      const distance = parseFloat(document.getElementById('distance').value);
      const time = parseFloat(document.getElementById('time').value);
      const output = document.getElementById('output');

      if (!price || !distance || !time) {
        output.textContent = "Please fill in all fields.";
        output.className = "result reject";
        return;
      }

      const perKm = price / distance;
      const perMin = price / time;
      const meetsKm = perKm >= minPerKm;
      const meetsMin = perMin >= minPerMin;

      output.innerHTML = `
        EGP/km: ${perKm.toFixed(2)}<br>
        EGP/min: ${perMin.toFixed(2)}<br>
        Decision: ${meetsKm && meetsMin ? "‚úÖ Accept" : "‚ùå Reject"}
      `;
      output.className = "result " + (meetsKm && meetsMin ? "accept" : "reject");
    }

    if ('serviceWorker' in navigator) {
      navigator.serviceWorker.register('service-worker.js');
    }
  </script>
