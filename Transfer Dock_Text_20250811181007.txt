  <p>Compare order prices from Didi and Uber.</p>

  <label>Didi price (EGP): <input type="number" id="didi"></label><br><br>
  <label>Uber price (EGP): <input type="number" id="uber"></label><br><br>

  <button onclick="checkBetter()">Check Best</button>

  <p id="result"></p>

  <script>
    function checkBetter() {
      const didi = parseFloat(document.getElementById('didi').value);
      const uber = parseFloat(document.getElementById('uber').value);

      if (isNaN(didi) || isNaN(uber)) {
        document.getElementById('result').innerText = "Enter both prices!";
        return;
      }

      if (didi < uber) {
        document.getElementById('result').innerText = "Didi is cheaper!";
      } else if (uber < didi) {
        document.getElementById('result').innerText = "Uber is cheaper!";
      } else {
        document.getElementById('result').innerText = "Both are the same price.";
      }
    }
  </script>
